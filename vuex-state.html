改好全部長這樣 
< div id="app">
  < counter>< /counter>
  < p>
    < button @click="increment">+< /button>
    < button @click="decrement">-< /button>
  < /p>
< /div>
< script type="text/javascript">
  // 创建一个 Counter 组件
  const store = new Vuex.Store({
    state: {
      count: 0
    }
  })

  const Counter = {
    template: `< div>{{ count }}< /div>`,
    computed: {
      count() {
        return this.$store.state.count
      }
    }
  }

  new Vue({
    el: '#app',
    store,
    components: {
      'counter': Counter
    },
    methods: {
      increment () {
        store.state.count++
      },
      decrement () {
        store.state.count--
      }
    }
  })
< /script>
